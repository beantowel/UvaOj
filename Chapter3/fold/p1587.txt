#include <stdio.h>
#include <string.h>
int a,b,c;
int l[2][6],count[3],match[3][2],sum;
void Count(int i);
void Sort(int *a,int *b);
int main(){
	while(scanf("%d %d\n",&l[0][0],&l[1][0])!=EOF){
		sum=0;
		for(int i=1;i<6;i++){
			scanf("%d %d\n",&l[0][i],&l[1][i]);
			sum+=l[0][i];
			sum+=l[1][i];
		}
		sum=sum/4;
		a=l[0][0];
		b=l[1][0];
		c=sum-a-b;
		Sort(&a,&b);
		Sort(&a,&c);
		Sort(&b,&c);
		match[0][0]=a;match[0][1]=b;
		match[1][0]=a;match[1][1]=c;
		match[2][0]=b;match[2][1]=c;
		for(int i=0;i<6;i++)
			Count(i);
	}
	return 0;
}
void Count(int i){
	memset(count,0,sizeof(count));
	int x=l[0][i],y=l[1][i];
	int j=0;
	Sort(&x,&y);
	while((count[j]<2) && (j<3)){
		j++;
	}
}
void Sort(int *a,int*b){ //a<b
	if(a>b){
		int t=*a;
		*a=*b;
		*b=t;
	}
}
